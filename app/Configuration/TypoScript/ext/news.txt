plugin.tx_news {
  # Override Language
	_LOCAL_LANG.de{
		more-link = Hier mehr erfahren
	}

	view {
    # Override Widget Template Path
		widget.GeorgRinger\News\ViewHelpers\Widget\PaginateViewHelper.templateRootPath = {$plugin.tx_news.view.templateRootPath}
	}

  settings {
    # Disable Default CSS
    cssFile >

    # Teaser Max Chars
    cropMaxCharacters = 200

    # Disable Dummy Image
    displayDummyIfNoMedia = 0

    detail {
      # Disable Social Share Buttns
      showSocialShareButtons = 0
      media {
        image {
          # set Fancybox as lightbox
          lightbox = fancybox

          # Override Image width
          maxWidth = 800
        }
      }
    }
    list {
        media {
            image {
                # Override Image Sizes
                maxWidth = 480
                maxHeight = 480
            }
        }
        #Paginate configuration.
        paginate {
            insertAbove = 0
            insertBelow = 1
        }
    }
  }
}

###########################
### Add News to Sitemap ###
###########################
plugin.tx_seo.config {
    xmlSitemap {
        sitemaps {
            news {
                provider = TYPO3\CMS\Seo\XmlSitemap\RecordsXmlSitemapDataProvider
                config {
                    table = tx_news_domain_model_news
                    additionalWhere =
                    sortField = sorting
                    lastModifiedField = tstamp
                    changeFreqField = sitemap_changefreq
                    priorityField = sitemap_priority
                    pid = {$plugin.tx_csseo.sitemap.extensions.news.storagePid}
                    recursive = 2
                    url {
                        pageId = {$plugin.tx_csseo.sitemap.extensions.news.detailPid}
                        fieldToParameterMap {
                            uid = tx_news_pi1[news]
                        }

                        additionalGetParameters {
                            tx_news_pi1.controller = News
                            tx_news_pi1.action = detail
                        }
                    }
                }
            }
        }
    }
}
